1 /**
  2  *
  3  * Reusable utility functions
  4  * 
  5  * Ideally, all the scripts you use would wrap themselves in a define() call so 
  6  * that RequireJS can understand what dependencies need to be loaded before 
  7  * defining the script's functionality.
  8  * 
  9  * @author Aki Karkkainen
 10  * @url https://github.com/akikoo/one-web-boilerplate
 11  * Twitter: http://twitter.com/akikoo
 12  * 
 13  **/
 14 
 15 
 16 define([
 17     // Libraries
 18     'jquery'
 19 ],
 20 
 21 
 22 function ($) {
 23 
 24     'use strict';
 25 
 26     var utils = {
 27 
 28         /**
 29          * @author Aki Karkkainen
 30          * @description Loads stylesheets.
 31          * @param {String} url Stylesheet url.
 32          * @example utils.loadCSS('/assets/css/new.css');
 33          * @returns N/A
 34          */
 35         loadCSS: function (url) {
 36 
 37             var d   = document,
 38                 s   = d.createElement('link'),
 39                 x   = d.getElementsByTagName('head')[0];
 40 
 41             s.rel   = 'stylesheet';
 42             s.href  = url;
 43             x.appendChild(s);
 44         },
 45 
 46 
 47         /**
 48          * @author Aki Karkkainen
 49          * @description Creates script tags dynamically via JavaScript. 
 50          * Non-blocking download begins immediately, and the script executes 
 51          * as soon as it is downloaded completely.
 52          * @param {String} url Script url.
 53          * @example utils.loadJS('/assets/js/modules/new.js');
 54          * @returns N/A
 55          */
 56         loadJS: function (url) {
 57 
 58             var d   = document,
 59                 js  = d.createElement('script');
 60 
 61             js.src  = url;
 62             (d.head || d.getElementsByTagName('head')[0]).appendChild(js);
 63         },
 64 
 65 
 66         /**
 67          * @author Aki Karkkainen. From JavaScript Patterns by Stoyan Stefanov.
 68          * @description Checks for Internet Explorer browser (all IE versions) 
 69          * using conditional comments. Useful if we only need to detect IE in 
 70          * general, without a version number. It is slightly safer than looking 
 71          * for strings in navigator.userAgent, because these strings are easy to 
 72          * change by the user.
 73          * @example utils.isIE();
 74          * @returns {Boolean} Will return false in all browsers (because they 
 75          * ignore the comment), but it will be true in Internet Explorer, 
 76          * because of the negation ! in the conditional comment.
 77          */
 78         isIE: function () {
 79             // IE returns !false;
 80             return /*@cc_on!@*/false;
 81         },
 82 
 83 
 84         /**
 85          * @author Aki Karkkainen
 86          * @description Checks for specific Internet Explorer browser version 
 87          * from conditional comments.
 88          * @param {String} ieversion IE version string from conditional comments. 
 89          * (ie6, ie7, iem7, ie8, ie9).
 90          * @example utils.isIEVersion(ie7);
 91          * @returns {Boolean} On IE, returns true for IE version passed in as a parameter.
 92          */
 93         isIEVersion: function (ieversion) {
 94             return document.documentElement.className.indexOf(ieversion) >= 0;
 95         }
 96 
 97     };
 98 
 99     return utils;
100 
101 });
